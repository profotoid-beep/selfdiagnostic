let state = {

  answers: {},

  currentQuestion: 0

};


const STORAGE_KEY = 'tochka_sborki_v1';


function saveState() {

  try { localStorage.setItem(STORAGE_KEY, JSON.stringify(state)); } catch(e) {}

}


function loadState() {

  try {

    const saved = localStorage.getItem(STORAGE_KEY);

    if (saved) {

      const parsed = JSON.parse(saved);

      if (parsed && typeof parsed === 'object') state = parsed;

    }

  } catch(e) {}

}


function resetState() {

  state = { answers: {}, currentQuestion: 0 };

  try { localStorage.removeItem(STORAGE_KEY); } catch(e) {}

}

/* ============================================

   ANSWER HANDLERS

============================================ */

function selectChoice(qId, letter) {

  state.answers[qId] = letter;

  const screen = document.getElementById(`screen-q${qId}`);

  screen.querySelectorAll('.option').forEach(o => {

    o.classList.toggle('selected', o.dataset.letter === letter);

  });

  document.getElementById(`next-btn-${qId}`).disabled = false;

  saveState();

}


function updateScale(qId, value) {

  state.answers[qId] = parseInt(value);

  const q = QUESTIONS.find(x => x.id === qId);

  document.getElementById(`scale-value-${qId}`).textContent = value;

  document.getElementById(`scale-caption-${qId}`).textContent = q.captions[value];

  document.getElementById(`next-btn-${qId}`).disabled = false;

  saveState();

}


function updateText(qId, value) {

  state.answers[qId] = value.trim();

  const q = QUESTIONS.find(x => x.id === qId);

  const len = value.trim().length;

  const counter = document.getElementById(`text-counter-${qId}`);

  counter.textContent = `${len} символов · минимум ${q.minLength}`;

  document.getElementById(`next-btn-${qId}`).disabled = len < q.minLength;

  saveState();

}


/* ============================================

   NAVIGATION

============================================ */

function showScreen(id) {

  document.querySelectorAll('.screen').forEach(s => s.classList.remove('active'));

  const el = document.getElementById(id);

  if (el) {

    el.classList.add('active');

    window.scrollTo({ top: 0, behavior: 'smooth' });

  }

}


function updateProgress() {

  const total = QUESTIONS.length;

  const current = state.currentQuestion;

  const fill = (current / total) * 100;

  document.getElementById('progressFill').style.transform = `scaleX(${fill / 100})`;

  document.getElementById('progressText').textContent = `${current} / ${total}`;

}


function startTest() {

  state.currentQuestion = 1;

  document.getElementById('header').classList.remove('hidden');

  showScreen('screen-q1');

  restoreAnswer(1);

  updateProgress();

  saveState();

}


function restoreAnswer(qId) {

  const ans = state.answers[qId];

  if (ans === undefined) return;

  const q = QUESTIONS.find(x => x.id === qId);

  if (q.type === 'choice') {

    const screen = document.getElementById(`screen-q${qId}`);

    screen.querySelectorAll('.option').forEach(o => {

      o.classList.toggle('selected', o.dataset.letter === ans);

    });

    document.getElementById(`next-btn-${qId}`).disabled = false;

  } else if (q.type === 'scale') {

    document.getElementById(`scale-input-${qId}`).value = ans;

    document.getElementById(`scale-value-${qId}`).textContent = ans;

    document.getElementById(`scale-caption-${qId}`).textContent = q.captions[ans];

    document.getElementById(`next-btn-${qId}`).disabled = false;

  } else if (q.type === 'text') {

    document.getElementById(`text-input-${qId}`).value = ans;

    const len = ans.length;

    document.getElementById(`text-counter-${qId}`).textContent = `${len} символов · минимум ${q.minLength}`;

    document.getElementById(`next-btn-${qId}`).disabled = len < q.minLength;

  }

}


function goNext() {

  if (state.currentQuestion < QUESTIONS.length) {

    state.currentQuestion++;

    if (state.currentQuestion <= QUESTIONS.length) {

      showScreen(`screen-q${state.currentQuestion}`);

      restoreAnswer(state.currentQuestion);

      updateProgress();

      saveState();

    }

  } else {

    finishTest();

  }


  // Если это был последний вопрос

  if (state.currentQuestion > QUESTIONS.length) {

    finishTest();

  }

}


function goPrev() {

  if (state.currentQuestion > 1) {

    state.currentQuestion--;

    showScreen(`screen-q${state.currentQuestion}`);

    restoreAnswer(state.currentQuestion);

    updateProgress();

    saveState();

  } else {

    state.currentQuestion = 0;

    document.getElementById('header').classList.add('hidden');

    showScreen('screen-intro');

    updateProgress();

    saveState();

  }

}


// Override goNext for last question

function checkLastQuestion() {

  // Replace last "Next" button behavior

  const lastBtn = document.getElementById(`next-btn-${QUESTIONS.length}`);

  if (lastBtn) {

    lastBtn.textContent = 'Получить разбор';

    lastBtn.onclick = finishTest;

  }

}


function finishTest() {

  document.getElementById('header').classList.add('hidden');

  showScreen('screen-processing');

  setTimeout(() => {

    renderReport();

    showScreen('screen-report');

  }, 3200);

}

function getPointType(a) {

  const q1 = a[1], q12 = a[12];

  if (q12 === 'Д') return 'crisis';

  if (q1 === 'Г' || q12 === 'Г') return 'transition';

  if (q1 === 'В') return 'anxious_drive';

  if (q1 === 'А' && q12 === 'В') return 'wrong_direction';

  if (q1 === 'А') return 'hidden_burnout';

  if (q1 === 'Б') return 'quiet_stagnation';

  return 'hidden_burnout';

}


function getBlockType(a) {

  return { 'А': 'readiness', 'Б': 'guarantee', 'В': 'time_race', 'Г': 'helplessness' }[a[6]] || 'readiness';

}


function getVector(a) {

  return { 'А': 'depth', 'Б': 'transformation', 'В': 'pause', 'Г': 'unblock', 'Д': 'self_encounter' }[a[10]] || 'self_encounter';

}


function getTechnique(point, block, readiness) {

  if (readiness >= 8) return 'questions';

  if (point === 'crisis' || point === 'transition') return 'future_image';

  if (point === 'anxious_drive') return 'letter';

  if (point === 'wrong_direction' && block === 'guarantee') return 'refusal';

  if (point === 'hidden_burnout' || point === 'wrong_direction') return 'leak_map';

  if (point === 'quiet_stagnation') return 'silence';

  return 'leak_map';

}

function escapeHTML(str) {

  return String(str)

    .replace(/&/g, '&amp;')

    .replace(/</g, '&lt;')

    .replace(/>/g, '&gt;')

    .replace(/"/g, '&quot;')

    .replace(/'/g, '&#039;');

}


function downloadPDF() {

  const element = document.getElementById('report-content');

  const opt = {

    margin:       [15, 15, 15, 15],

    filename:     'tochka-sborki.pdf',

    image:        { type: 'jpeg', quality: 0.98 },

    html2canvas:  { scale: 2, useCORS: true, backgroundColor: '#F4EFE6' },

    jsPDF:        { unit: 'mm', format: 'a4', orientation: 'portrait' },

    pagebreak:    { mode: ['avoid-all', 'css', 'legacy'] }

  };

  html2pdf().set(opt).from(element).save();

}
